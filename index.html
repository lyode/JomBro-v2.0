const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' }); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = `luna-conversation-${new Date().toISOString().split('T')[0]}.json`; a.click(); showNotification('Conversation exported', 'success'); }; // UI Components const MessageBubble = ({ message }) => { const isUser = message.type === 'user'; const emotionColors = { happy: 'border-green-500', sad: 'border-blue-500', neutral: 'border-gray-500', excited: 'border-yellow-500', concerned: 'border-red-500', friendly: 'border-purple-500' };
